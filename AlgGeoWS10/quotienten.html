\NeedsTeXFormat{LaTeX2e}
\ProvidesPackage{quotienten}[2011/08/09]

%Quotienten:

\newcommand*{\@old@slash}{}\let\@old@slash\slash
\def\slash{\relax\ifmmode\delimiter"502F30E\mathopen{}\else\@old@slash\fi}

\def\backslash{\delimiter"526E30F\mathopen{}}
\newcommand{\rquot}[2]{
  \mathchoice
  { %displaystyle
    \left.
    \raisebox{0.5ex}{\ensuremath{#1}}
    \middle\ensuremath\slash
    \raisebox{-0.4ex}{\ensuremath{#2}}
    \right.  
  }
  { %textstyle
    #1\ensuremath\slash #2
  }
  { %scriptstyle
    #1\ensuremath\slash #2
  }
  { %scriptscriptstyle
    #1\ensuremath\slash #2
  }
}
\newcommand{\lquot}[2]{
  \mathchoice
  { %displaystyle
    \left.
    \raisebox{-0.4ex}{\ensuremath{#2}}
    \middle\ensuremath\backslash
    \raisebox{0.5ex}{\ensuremath{#1}}
    \right.  }
  { %textstyle
    #2\ensuremath\backslash #1
  }
  { %scriptstyle
    #2\ensuremath\backslash #1
  }
  { %scriptscriptstyle
    #2\ensuremath\backslash #1
  }
}
\newcommand{\lrquot}[3]{
  \mathchoice
  { %displaystyle
    \left.
    \raisebox{-0.4ex}{\ensuremath{#2}}
    \middle\ensuremath\backslash
    \raisebox{0.5ex}{\ensuremath{#1}}
    \middle\ensuremath\slash
    \raisebox{-0.4ex}{\ensuremath{#3}}    
    \right.  }
  { %textstyle
    #2\ensuremath\backslash #1\ensuremath\slash #3
  }
  { %scriptstyle
    #2\ensuremath\backslash #1\ensuremath\slash #3
  }
  { %scriptscriptstyle
    #2\ensuremath\backslash #1\ensuremath\slash #3
  }
}

\newcommand{\Quotient}[2]{\rquot{#1}{#2}}
\endinput
